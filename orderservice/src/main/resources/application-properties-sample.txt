spring.application.name=orderservice
server.port=8084
# Kafka Broker
spring.kafka.bootstrap-servers=localhost:9092
# Topics (In which message to be received from)
#app.kafka.topic=${KAFKA_TOPIC}
app.kafka.order-created.topic=${KAFKA_TOPIC_1}
app.kafka.order-updated.topic=order-${KAFKA_TOPIC_2}

spring.kafka.consumer.group-id=${KAFKA_GROUP_ID}

spring.kafka.consumer.key-deserializer=org.apache.kafka.common.serialization.StringDeserializer
spring.kafka.consumer.value-deserializer=org.apache.kafka.common.serialization.StringDeserializer
# Offset Settings
spring.kafka.consumer.auto-offset-reset=earliest
# Database properties
# ===============================

# MySQL Database Configuration

# ===============================

# ===============================
